
Install :

sudo su

LC_ALL="en_US.UTF-8"
LC_CTYPE="en_US.UTF-8"
echo en_US.UTF-8 UTF-8 > /etc/locale.gen

apt install curl openssh-server ca-certificates postfix
curl -sS https://packages.gitlab.com/install/repositories/gitlab/gitlab-ce/script.deb.sh | bash
apt install gitlab-ce
gitlab-ctl reconfigure

curl -L https://packages.gitlab.com/install/repositories/runner/gitlab-runner/script.deb.sh | bash
apt-get install gitlab-runner
usermod -aG docker gitlab-runner

---
Registry :
docker run -d \
-p 5000:5000 --restart always --name registry \
-v /srv/registry:/var/lib/registry \
registry:2

---
Runner :
gitlab-runner verify --delete

gitlab-runner unregister \
  --url http://gitlab.example.com/ \
  --token j36dSBQt1pVnkx8yVuD_

gitlab-runner register -n --locked=false \
  --url http://gitlab.example.com/ \
  --registration-token j36dSBQt1pVnkx8yVuD_ \
  --tag-list [shell] \
  --executor shell \
  --name "runner-shell"

gitlab-runner register -n --locked=false \
  --name "runner-docker" \
  --url http://gitlab.example.com/ \
  --registration-token j36dSBQt1pVnkx8yVuD_ \
  --tag-list [docker] \
  --executor docker \
  --docker-image docker:latest \
  --docker-privileged true \
  --docker-network-mode gitlab_default \
  --docker-extra-hosts ["gitlab.example.com:192.168.33.10", "registry.gitlab.example.com:192.168.33.10"]

# List runners
gitlab-runner list

# Unit Run
gitlab-runner run
gitlab-runner --debug run

# Launch as service
gitlab-runner install root

# Docker in docker
docker run --privileged --name some-docker -d docker:dind
docker run -it --rm --link some-docker:docker docker version
docker run -it --rm --link some-docker:docker docker sh
